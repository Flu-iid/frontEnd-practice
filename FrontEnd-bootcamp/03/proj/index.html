function arrayMaker(start, end, step) {
    var array = [];
    for (var i = start; i <= end; i += step){
        array.push(i)
    }
    return array
}

function checkPrime (number) {
    if (number < 2) {
        return false;
      } else {
        for (var i = 2; i < number; i++) {
          if (number % i == 0) {
            return false;
          }
        }
        return true;
      }
}

var startVal = 1
var endVal = 1000
var stepVal = 1

var numbers = arrayMaker(startVal, endVal, stepVal)

var primeNumbers = numbers.filter(function(item){
    return checkPrime(item)
})

var sum = primeNumbers.reduce(function(content, y){
    return content+y
})

var count = primeNumbers.reduce(function(content, y){
    return content+=1
})
